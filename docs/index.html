    "0#{value.toString(16)}".slice(-2)

  hex.text text

toCanvas = (view, palette) ->
  chunkSize = 64
  pixelSize = 8

  [0...(view.length / chunkSize)].forEach (chunk) ->
    console.log "drawing chunk #{chunk}"
    [0...8].forEach (row) ->
      [0...8].forEach (col) ->
        index = view[chunk * chunkSize + row * 8 + col]
        if index > 16
          debugger

        x = chunkSize * (chunk % 10)
        y = chunkSize * (chunk / 10).floor()

        canvas.drawRect
          x: x + col * 8
          y: y + row * 8
          width: pixelSize
          height: pixelSize
          color: palette[index]
