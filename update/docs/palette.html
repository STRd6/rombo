numberToHex = (n) ->
  "0#{n.toString(0x10)}".slice(-2).toUpperCase()

makeHexColors = (lines) ->
  lines.split("\n").map (line) ->
    "#" + line.split(" ").map (string) ->
      numberToHex parseInt(string, 10)
    .join("")

module.exports =
  default4Color: makeHexColors """
    20 12 28

  "15-bit RGB": (word) ->
    R = 0b0000000000011111
    G = 0b0000001111100000
    B = 0b0111110000000000

    r = word & R
    g = (word & G) >> 5
    b = (word & B) >> 10

    "rgb(#{r}, #{g}, #{b})"
